#### `core/addRow/index.ts`

```ts
import type { PlankType, PoseParams, Room, Row } from '@/core/types'
import { intersectStripExtents } from '@/core/geometry'
import { canAppendRow, computeRowYStart } from '@/core/rowYStart'
import { computeDefaultXOffset } from './computeDefaultXOffset'

export { computeDefaultXOffset } from './computeDefaultXOffset'

export function addRow(
  room: Room,
  plankType: PlankType,
  poseParams: PoseParams,
  catalog: PlankType[],
): Row | null {
  if (!canAppendRow(room, catalog, poseParams)) return null

  const yStart = computeRowYStart(room, room.rows.length, catalog, poseParams)
  const yEnd = yStart + plankType.width

  const xSegments = intersectStripExtents(room.vertices, yStart, yEnd)
  const firstXOffset = computeDefaultXOffset(room, plankType, poseParams, undefined, catalog)

  const segments = xSegments.length > 0
    ? xSegments.map((_, i) => ({ xOffset: i === 0 ? firstXOffset : 0 }))
    : [{ xOffset: firstXOffset }]

  return {
    id: crypto.randomUUID(),
    roomId: room.id,
    plankTypeId: plankType.id,
    segments,
  }
}
```

#### `useDrawInteraction.helpers.ts`

```ts
import type { Point } from '@/core/types'
import { findSnap } from '@/core/snap'

const FORM_SELECTOR = 'input, textarea, select, [contenteditable]'

export function focusInsideForm(): boolean {
  const el = document.activeElement as HTMLElement | null
  return !!el && typeof el.matches === 'function' && el.matches(FORM_SELECTOR)
}

export function placeVertex(
  world: Point, ctrl: boolean, vertices: Point[], zoom: number, rootFontPx: number,
): Point {
  if (!ctrl) return { x: world.x, y: world.y }
  const r = findSnap(world, vertices, zoom, rootFontPx)
  return { x: r.x, y: r.y }
}

export function shouldCloseOnFirst(
  world: Point, vertices: Point[], zoom: number, thresholdPx: number,
): boolean {
  if (vertices.length < 3) return false
  const first = vertices[0]
  const dx = (first.x - world.x) * zoom
  const dy = (first.y - world.y) * zoom
  return Math.hypot(dx, dy) <= thresholdPx
}
```

`focusInsideForm` touche au DOM → reste co-localisé au hook (pas dans `core/`). Les deux autres fonctions (`placeVertex`, `shouldCloseOnFirst`) sont **pures** mais restent dans `helpers.ts` plutôt que `core/` car spécifiques à l'interaction de dessin (pas réutilisées).

Le hook `useDrawInteraction` importe ces helpers et perd ~40 LOC, descendant à ~115 LOC.

#### `usePlankTypeFormState.ts`

```ts
import { useState, useMemo } from 'react'
import type { PlankType, PlankTypePricing } from '@/core/types'

type PricingType = 'unit' | 'lot'

const parseNum = (s: string): number => {
  const n = parseFloat(s.replace(',', '.'))
  return Number.isFinite(n) ? n : NaN
}

export function usePlankTypeFormState(initial: PlankType | null) {
  const [name, setName] = useState(initial?.name ?? '')
  const [length, setLength] = useState(String(initial?.length ?? 120))
  const [width, setWidth] = useState(String(initial?.width ?? 14))
  const [pricingType, setPricingType] = useState<PricingType>(initial?.pricing.type ?? 'unit')
  const [priceUnit, setPriceUnit] = useState(
    initial?.pricing.type === 'unit' ? String(initial.pricing.pricePerUnit) : '0',
  )
  const [priceLot, setPriceLot] = useState(
    initial?.pricing.type === 'lot' ? String(initial.pricing.pricePerLot) : '0',
  )
  const [lotSize, setLotSize] = useState(
    initial?.pricing.type === 'lot' ? String(initial.pricing.lotSize) : '1',
  )
  const [description, setDescription] = useState(initial?.description ?? '')

  const lengthN = parseNum(length)
  const widthN = parseNum(width)
  const valid = useMemo(() => (
    name.trim().length > 0 && lengthN > 0 && widthN > 0
    && (pricingType === 'unit'
      ? parseNum(priceUnit) >= 0
      : parseNum(priceLot) >= 0 && parseNum(lotSize) >= 1)
  ), [name, lengthN, widthN, pricingType, priceUnit, priceLot, lotSize])

  const buildPricing = (): PlankTypePricing =>
    pricingType === 'unit'
      ? { type: 'unit', pricePerUnit: parseNum(priceUnit) }
      : { type: 'lot', pricePerLot: parseNum(priceLot), lotSize: parseNum(lotSize) }

  return {
    fields: { name, length, width, pricingType, priceUnit, priceLot, lotSize, description },
    setters: { setName, setLength, setWidth, setPricingType, setPriceUnit, setPriceLot, setLotSize, setDescription },
    valid,
    lengthN,
    widthN,
    buildPricing,
  }
}
```

## Vérification manuelle

Smoke test complet de l'app, dernière passe avant merge `main`. Périmètre minimal mais exhaustif sur les fonctions critiques :

1. **Création projet** : `/new` → scratch → projet vide créé. URL `/{id}/main`.
2. **Import JSON** : `/new` → import → projet chargé. Catalogue, pièces, rangées, plan de fond restitués.
3. **Pose** : modifier `cale`, `minPlankLength`, `minRowGap`. Recalcul live des pièces existantes.
4. **Catalogue** : ajouter un type, lien URL si description, tooltip. Modifier (length/width verrouillés si en usage). Supprimer (bloqué si en usage).
5. **Pièces** :
   - Mode `draw` : tracer un polygone, snap Ctrl, fermeture sur le premier sommet, dialog nom.
   - Mode `main` : drag d'un sommet (preview live, commit au release). Shift = sommets adjacents. Ctrl = snap.
   - yOffset par pièce : modifier la valeur, recalcul live.
6. **Rangées** :
   - Ajouter une rangée du type sélectionné.
   - Drag d'un segment (preview, commit, cascade offcut sur segment[0] uniquement).
   - Édition inline `firstLength` / `lastLength` (annotations).
   - Bulk fill rangées du drawer.
   - Suppression d'une rangée.
7. **Drawer résumé** : ouvrir, vérifier tableau matière (col Lames sans quantité), tableau réutilisations, hover→highlight. Export CSV : 3 sections (matière, détail par pièce, liens), triplets par segment, règles colonnes vides.
8. **Plan de fond** : import image, repositionnement (drag), calibration. Mode `plan`.
9. **Undo/Redo** : Ctrl+Z après un drag → rollback. Ctrl+Y → forward. Drafts en mode draw : Ctrl+Z supprime le dernier point local, pas un undo Redux.
10. **Routing** : copier URL, recharger → état restauré. Projet inexistant → `/new`. Pièce inexistante → `/{id}/main`.
11. **Theme** : switcher dark/light/system.
12. **Persistance** : recharger l'app → projet restauré (IDB v4).

Si **un seul** point échoue, stopper et investiguer avant merge.

- **Risque n°1 — barrels qui cassent les imports** : `selectors/index.ts` et `projectSlice/index.ts` doivent ré-exporter **exactement** les symboles publics existants. Vérifier avant commit :
  ```bash
  diff <(grep -rh "from '@/store/selectors'" src/ | sort -u) <(grep -rh "from '@/store/selectors'" src/ | sort -u)
  ```
  (avant/après refacto, mais avec staging git). Si un consommateur ne compile plus → ajouter le ré-export manquant.

- **Risque n°2 — Immer + reducers extraits** : `createSlice` accepte un objet de reducers ; passer un spread d'objets fonctionne tant que chaque reducer prend `state: ProjectState` (l'arg `state` est déjà un draft Immer).
